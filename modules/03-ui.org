    (defun taan/load-tokyo-theme (&optional frame)
      (with-selected-frame (or frame (selected-frame))
        (load-theme 'doom-tokyo-night t)
        (set-face-foreground 'line-number-current-line "#ff9e64")
        (taan/apply-tokyo-modeline)
        (run-at-time 0.5 nil #'taan/apply-tokyo-modeline))
      (remove-hook 'after-make-frame-functions #'taan/load-tokyo-theme))

    (use-package doom-themes
    :config
    (setq doom-themes-enable-bold t
          doom-themes-enable-italic t)
    (if (daemonp)
        (add-hook 'after-make-frame-functions #'taan/load-tokyo-theme)
      (taan/load-tokyo-theme)))

    (use-package doom-modeline
    :init
    (doom-modeline-mode 1)
    :config
    (setq doom-modeline-battery t
          doom-modeline-icon t
          doom-modeline-major-mode-icon t
          doom-modeline-major-mode-color-icon t))
